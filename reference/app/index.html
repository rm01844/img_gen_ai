    def login_required(f):
    """Decorator that restricts access to logged-in admin users."""
    from functools import wraps

    @wraps(f)
    def decorated_function(*args, **kwargs):
        if not session.get("logged_in"):
            flash("Please log in to access the generator.", "warning")
            return redirect(url_for("login"))
        return f(*args, **kwargs)
    return decorated_function

    @app.route("/generate", methods=["POST"])
def generate_image() -> Response:
    """Generate image(s) using Vertex AI Imagen model.

    This endpoint generates one or more images based on the user's prompt and options.

    Request Body (JSON):
        prompt (str): The image description to generate.
        number_of_images (int, optional): Number of images to generate (default=1).
        aspect_ratio (str, optional): Aspect ratio, e.g. "1:1" or "16:9".
        negative_prompt (str, optional): Objects/concepts to avoid.

    Returns:
        Response: JSON containing a list of image URLs or an error message.
    """
    try:
        data = request.get_json()
        prompt = data.get("prompt")
        number_of_images = data.get("number_of_images", "1")
        aspect_ratio = data.get("aspect_ratio", "1:1")
        negative_prompt = data.get("negative_prompt", "")

        if not prompt:
            return jsonify({"error": "No prompt provided"}), 400

        # Load Imagen model
        model = ImageGenerationModel.from_pretrained("imagen-4.0-generate-001")

        # Generate image
        result = model.generate_images(
            prompt=prompt,
            number_of_images=int(number_of_images),
            aspect_ratio=aspect_ratio,
            negative_prompt=negative_prompt,
            person_generation="allow_all",
            safety_filter_level="block_few",
            add_watermark=True,
        )

        # Save image to static folder
        image_urls = []
        for img in result.images:
            static_dir = os.path.join(os.path.dirname(__file__), "static")
            os.makedirs(static_dir, exist_ok=True)

            filename = f"generated_{uuid.uuid4().hex}.png"
            output_path = os.path.join(static_dir, filename)
            img.save(output_path)

            filename = os.path.basename(output_path)
            image_urls.append(f"/static/{filename}?v={int(time.time())}")

        # Return image URL with timestamp (cache-buster)
        return jsonify({"image_urls": image_urls})

    except Exception as e:
        print("Error:", str(e))
        return jsonify({"error": str(e)}), 500

    def refine_prompt(raw_prompt: str, reference_description: str = "") -> str:
    """Use Gemini 2.5 Pro to rewrite user prompt for Imagen clarity."""

    def generate_with_imagen(prompt: str, img_b64: str, num_images: int = 1) -> list[str]:
    """Send a request to Vertex Imagen and return generated image URLs."""
    body = {
        "instances": [
            {
                "prompt": prompt,
                "referenceImages": [
                    {
                        "referenceType": "REFERENCE_TYPE_RAW",
                        "referenceId": 1,
                        "referenceImage": {
                            "bytesBase64Encoded": img_b64
                        },
                    }
                ],
            }
        ],
        "parameters": {"sampleCount": num_images},
    }

    headers = {
        "Authorization": f"Bearer {credentials.token}",
        "Content-Type": "application/json",
    }

    response = requests.post(ENDPOINT, headers=headers, json=body)
    response.raise_for_status()
    result = response.json()

    image_urls = []
    static_dir = os.path.join(os.path.dirname(__file__), "static")
    os.makedirs(static_dir, exist_ok=True)

    for pred in result.get("predictions", []):
        data_b64 = pred.get("bytesBase64Encoded")
        if not data_b64:
            continue
        filename = f"edited_{uuid.uuid4().hex}.png"
        output_path = os.path.join(static_dir, filename)
        with open(output_path, "wb") as f:
            f.write(base64.b64decode(data_b64))
        image_urls.append(f"/static/{filename}?v={int(time.time())}")

    return image_urls

    def detect_face_bbox(image_path: str) -> Tuple[int, int, int, int] | None:
    """Return (x, y, w, h) face bbox using OpenCV Haar as a lightweight fallback."""
    img = cv2.imread(image_path)
    if img is None:
        return None
    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    # Built-in cascade (simple but robust enough for crop)
    face_cascade = cv2.CascadeClassifier(
        cv2.data.haarcascades + 'haarcascade_frontalface_default.xml')
    faces = face_cascade.detectMultiScale(gray, 1.2, 5, minSize=(80, 80))
    if len(faces) == 0:
        return None
    # choose largest face
    faces = sorted(faces, key=lambda b: b[2]*b[3], reverse=True)
    return tuple(int(v) for v in faces[0])

    def crop_face(image_path: str) -> str | None:
    """Save a padded face crop next to the temp image and return its path."""
    img = cv2.imread(image_path)
    if img is None:
        return None
    bbox = detect_face_bbox(image_path)
    if not bbox:
        return None
    x, y, w, h = bbox
    # pad box
    pad = int(0.35 * max(w, h))
    x0 = max(0, x - pad)
    y0 = max(0, y - pad)
    x1 = min(img.shape[1], x + w + pad)
    y1 = min(img.shape[0], y + h + pad)
    face = img[y0:y1, x0:x1]
    face_path = os.path.join(tempfile.gettempdir(),
                             f"face_{uuid.uuid4().hex}.png")
    cv2.imwrite(face_path, face)
    return face_path

    def build_structured_prompt(user_prompt: str) -> str:
    """Normalize and structure the prompt into sections to improve instruction following."""
    # You already refine with Gemini; this complements it with layout scaffolding.
    lines = [ln.strip() for ln in user_prompt.split("\n") if ln.strip()]
    prompt = (
        "TASK: Edit the reference image while preserving the person's identity and clothing.\n"
        "LAYOUT: Respect a single, centered packaging with arranged accessories.\n"
        "CONTENT:\n"
        + "\n".join([f"- {ln}" for ln in lines]) +
        "\nSTYLE: studio lighting, product photo, realistic materials, sharp focus.\n"
        "TEXT: Ben Philip as top label (clear, sharp, correctly spelled).\n"
    )
    return prompt

    def restore_face_if_available(image_path: str) -> str:
    """Lightweight ONNX-based face enhancement using InsightFace (no Torch)."""
    global _face_app
    try:
        if _face_app is None:
            # Initialize only once
            _face_app = FaceAnalysis(name="buffalo_l", providers=[
                                     "CPUExecutionProvider"])
            _face_app.prepare(ctx_id=0)

        img = cv2.imread(image_path)
        if img is None:
            raise ValueError("Cannot read input image")

        faces = _face_app.get(img)
        if not faces:
            print("‚ö†Ô∏è No face detected ‚Äî skipping enhancement.")
            return image_path

        # Simple enhancement: blend detected face region for smoother texture
        for f in faces:
            box = f.bbox.astype(int)
            x1, y1, x2, y2 = box
            roi = img[y1:y2, x1:x2]
            if roi.size == 0:
                continue
            # Apply mild bilateral filtering + contrast boost
            smooth = cv2.bilateralFilter(roi, 9, 75, 75)
            enhanced = cv2.addWeighted(smooth, 0.6, roi, 0.4, 8)
            img[y1:y2, x1:x2] = enhanced

        output_path = image_path.replace(".png", "_restored.png")
        cv2.imwrite(output_path, img)
        return output_path

    except Exception as e:
        print(f"‚ö†Ô∏è Skipping restoration: {e}")
        return image_path

    @app.route("/edit", methods=["POST"])
def edit_image() -> Response:
    """Edit an uploaded image using Gemini + Imagen hybrid workflow.

    This endpoint accepts:
      - An uploaded image file
      - A prompt describing desired changes
      - Optional number_of_images parameter

    It performs:
      1Ô∏è‚É£ Prompt refinement via Gemini 2.5 Pro
      2Ô∏è‚É£ Image editing via Vertex AI Imagen 3
      3Ô∏è‚É£ Gemini Vision feedback + re-generation
    """
    try:
        raw_prompt = request.form.get(
            "prompt", "").strip() or "Modify this image"
        num_images = int(request.form.get("number_of_images", 1))

        if "image" not in request.files:
            return jsonify({"error": "No image uploaded"}), 400

        uploaded = request.files["image"]
        temp_path = os.path.join(
            tempfile.gettempdir(), f"upload_{uuid.uuid4().hex}.png")
        uploaded.save(temp_path)

        # üß† 1Ô∏è‚É£ Gemini refinement
        prompt = refine_prompt(raw_prompt)

        # üß† 2Ô∏è‚É£ Base64 encode the image
        with open(temp_path, "rb") as f:
            full_b64 = base64.b64encode(f.read()).decode("utf-8")

        # üß† 3Ô∏è‚É£ Construct Imagen-3-compatible request
        body = {
            "instances": [
                {
                    "prompt": prompt,
                    "image_bytes_base64": full_b64
                }
            ],
            "parameters": {
                "sampleCount": num_images
            }
        }

        credentials.refresh(Request())
        headers = {
            "Authorization": f"Bearer {credentials.token}",
            "Content-Type": "application/json",
        }

        print("üîç Sending payload keys:", list(body["instances"][0].keys()))

        resp = requests.post(ENDPOINT, headers=headers, json=body, timeout=120)
        print("üîç Status:", resp.status_code)
        if resp.status_code != 200:
            print("üîç Raw Response:", resp.text)
            return jsonify({"error": f"Vertex returned {resp.status_code}: {resp.text}"}), 400

        result = resp.json()
        preds = result.get("predictions", [])

        if not preds:
            return jsonify({"error": "No predictions returned."}), 400

        static_dir = os.path.join(os.path.dirname(__file__), "static")
        os.makedirs(static_dir, exist_ok=True)
        urls = []

        for pred in preds:
            b64 = pred.get("bytesBase64Encoded")
            if not b64:
                continue
            fn = f"edited_{uuid.uuid4().hex}.png"
            output_path = os.path.join(static_dir, fn)
            with open(output_path, "wb") as f:
                f.write(base64.b64decode(b64))
            output_path = restore_face_if_available(output_path)
            urls.append(
                f"/static/{os.path.basename(output_path)}?v={int(time.time())}")

        return jsonify({"image_urls": urls})

    except Exception as e:
        print("‚ùå Imagen Edit Error:", e)
        return jsonify({"error": str(e)}), 50
